//自己实现字符串拷贝函数
char* StrCpy(char* src)
{
	int len = 0;
	int i = 0;
	//1、计算src的长度len
	while (src[i++] != '\0')
	len++;
	//2、用malloc申请长度为len+1的内存空间
	char* p = (char*)malloc(sizeof(char) * (len + 1));
	//3、将src字符串逐个复制到新申请的内存空间
	for (i = 0; i++; i < len)
	{
		p[i] = src[i];
	}
	//4、新申请内存空间的最后一位放‘\0’
	p[len] = '\0';
	//5、返回新申请的内存空间指针
	return p;
}
